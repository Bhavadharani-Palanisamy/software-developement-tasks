#include <stdio.h>

#define SIZE 100

int data[SIZE];
int count = 0;

void create(int value) {
    if (count < SIZE) {
        data[count++] = value;
        printf("Added %d\n", value);
    } else {
        printf("Array is full!\n");
    }
}

void read() {
    printf("Current Data:\n");
    for (int i = 0; i < count; i++) {
        printf("%d ", data[i]);
    }
    printf("\n");
}

void update(int index, int newValue) {
    if (index >= 0 && index < count) {
        data[index] = newValue;
        printf("Updated index %d to %d\n", index, newValue);
    } else {
        printf("Invalid index!\n");
    }
}

void delete(int index) {
    if (index >= 0 && index < count) {
        for (int i = index; i < count - 1; i++) {
            data[i] = data[i + 1];
        }
        count--;
        printf("Deleted item at index %d\n", index);
    } else {
        printf("Invalid index!\n");
    }
}

int main() {
    create(10);
    create(20);
    create(30);
    read();

    update(1, 25);
    read();

    delete(0);
    read();
return 0;
}
