#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    int secret, guess;

    // Seed the random number generator
    srand(time(NULL));
    secret = rand() % 10 + 1; // Random number between 1 and 10

    printf("Guess a number between 1 and 10: ");
    scanf("%d", &guess);

    // Game logic using conditional statements
    if (guess == secret) {
        printf("🎉 Correct! You guessed the number.\n");
    } else if (guess < secret) {
        printf("Too low! Try again.\n");
    } else {
        printf("Too high! Try again.\n");
    }

    return 0;
}
